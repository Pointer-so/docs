## Prerequisites

* Node.js 16.0 or later

* A React-based application

## Installation

<Steps>
  <Step title="Grab an API key from the Pointer dashboard.">
    Login to the [Pointer dashboard](https://app.pointer.so), then create a project. Two keys will be automatically generated for you â€” a production and development key. Copy the development key and store it.
  </Step>

  <Step title="Install the Pointer SDK using your preferred package manager.">
    <CodeGroup>
      ```bash npm
      npm install @pointer-so/sdk
      ```

      ```bash pnpm
      pnpm add @pointer-so/sdk
      ```

      ```bash yarn
      yarn add @pointer-so/sdk
      ```
    </CodeGroup>
  </Step>

  <Step title="Configure your environment variables.">
    Add the Pointer API key into your `.env` file.

    ```bash
    POINTER_API_KEY=pt_dev_*********************
    ```
  </Step>

  <Step title="Wrap your app with the PointerProvider.">
    In your client-side root file, (e.g., `layout.tsx`, `index.tsx`, `_app.tsx`) wrap your app with the `PointerProvider`.

    ```typescript
    import { PointerProvider } from '@pointer-so/sdk';

    function App() {
      return (
        <PointerProvider apiKey={POINTER_API_KEY}>
          <YourApp />
        </PointerProvider>
      );
    }
    ```
  </Step>

  <Step title="Test the integration.">
    The Pointer widget should now appear in the bottom-right corner of your app. Head to the ['Developers' page](https://app.pointer.so/developers) in the dashboard and click "Verify connection" to ensure everything is working correctly.
  </Step>
</Steps>

## Framework-specific setup

<Note>
  Pointer currently works with any React-based application. Support for Vue, Svelte, and other frontend frameworks is in development.
</Note>

### Next.js

If using the `app` router:

```typescript app/layout.tsx
import { PointerProvider } from '@pointer-so/sdk';

export default function RootLayout({
  children,
}: {
  children: React.ReactNode;
}) {
  return (
    <html>
      <body>
        <PointerProvider 
          apiKey={process.env.NEXT_PUBLIC_POINTER_API_KEY!}
        >
          {children}
        </PointerProvider>
      </body>
    </html>
  );
}
```

If using the `pages` router:

```typescript pages/_app.tsx
import { PointerProvider } from '@pointer-so/sdk';
import type { AppProps } from 'next/app';

export default function App({ Component, pageProps }: AppProps) {
  return (
    <PointerProvider 
      apiKey={process.env.NEXT_PUBLIC_POINTER_API_KEY!}
    >
      <Component {...pageProps} />
    </PointerProvider>
  );
}
```

### Remix

```typescript app/root.tsx
import { PointerProvider } from '@pointer-so/sdk';

export default function App() {
  return (

        <PointerProvider
```
